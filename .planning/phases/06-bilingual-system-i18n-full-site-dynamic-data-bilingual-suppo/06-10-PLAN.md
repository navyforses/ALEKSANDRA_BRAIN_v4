        def redact_bilingual(
            pair: dict[str, str],
            consent: ConsentFlags,
        ) -> dict:
            """Run redact() on both halves of a {en, ka} pair and OR-block.

            Returns:
                {
                  "en": RedactionResult,
                  "ka": RedactionResult,
                  "blocked_or": bool,                 # True if either half blocked
                  "blocked_reasons": list[str],       # list of block_reason strings for blocked halves
                }

            Callers (agents/communicator.py, scripts/manager/briefing.py per 06-09 Task 3)
            must check blocked_or and raise BEFORE persisting.
            """
            en_result = redact(pair.get("en", ""), consent)
            ka_result = redact(pair.get("ka", ""), consent)
            blocked_or = en_result.blocked or ka_result.blocked
            blocked_reasons = []
            if en_result.blocked:
                blocked_reasons.append(f"en: {en_result.block_reason}")
            if ka_result.blocked:
                blocked_reasons.append(f"ka: {ka_result.block_reason}")
            return {
                "en": en_result,
                "ka": ka_result,
                "blocked_or": blocked_or,
                "blocked_reasons": blocked_reasons,
            }

        """I18N-10 — PHI redactor must survive into Georgian content."""
        from __future__ import annotations
        import yaml
        import pytest
        from pathlib import Path

        from scripts.communicator.phi_redactor import (
            redact, redact_bilingual, ConsentFlags, RedactionResult,
        )

        FIXTURE_PATH = Path(__file__).resolve().parent / "fixtures" / "phase6" / "phi_ka.yaml"

        with open(FIXTURE_PATH, encoding="utf-8") as f:
            FIXTURES = yaml.safe_load(f)

        assert len(FIXTURES) == 10, f"expected 10 Georgian PHI fixtures, got {len(FIXTURES)}"

        @pytest.mark.parametrize("fixture", FIXTURES, ids=lambda f: f["name"])
        def test_georgian_phi_redaction(fixture):
            """Each fixture's must_not_appear_in_output strings must be absent from the redacted text."""
            result: RedactionResult = redact(fixture["input"], ConsentFlags())

            if fixture.get("must_block"):
                assert result.blocked, f"{fixture['name']}: expected blocked=True"
                return

            for forbidden in fixture.get("must_not_appear_in_output", []):
                assert forbidden not in result.text, (
                    f"{fixture['name']}: forbidden token '{forbidden}' "
                    f"survived redaction. Output: {result.text!r}"
                )

        # ---------- redact_bilingual smoke tests ----------

                ConsentFlags(),

            assert "Aleksandra" not in r["en"].text
