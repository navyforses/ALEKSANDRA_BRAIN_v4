        """I18N-10 — imperative-verb lint must be bilingual-clean across 30 sample digests."""
        from __future__ import annotations
        import json
        from pathlib import Path
        import pytest
        from scripts.communicator.banned_phrases import check

        FIXTURE_PATH = Path(__file__).resolve().parent / "fixtures" / "phase6" / "bilingual_samples.json"
        with open(FIXTURE_PATH, encoding="utf-8") as f:
            SAMPLES = json.load(f)
        assert len(SAMPLES) == 30, f"expected 30 bilingual samples, got {len(SAMPLES)}"

        @pytest.mark.parametrize("sample", SAMPLES, ids=lambda s: s["id"])
        def test_bilingual_sample_lint_clean(sample):
            en_result = check(sample["en"], locales=("en",))
            assert en_result.passed, f"{sample['id']} en: {en_result.violations}"
            ka_result = check(sample["ka"], locales=("ka",))
            assert ka_result.passed, f"{sample['id']} ka: {ka_result.violations}"

        POSITIVE_KA_CASES = [
            ("ka-should",   "მართებთ რომ მიიღოს თერაპია"),

            ("ka-must-pred","აუცილებელია მკურნალობის გაგრძელება"),
            ("ka-consider", "განიხილეთ ეს ვარიანტი"),
            ("ka-consider2","გაითვალისწინეთ ეს რეკომენდაცია"),
            ("ka-try",      "სცადეთ ეს მიდგომა"),
            ("ka-askfor",   "მოითხოვეთ კონსულტაცია"),
            ("ka-request",  "ითხოვეთ მეორე აზრი"),
        ]

        @pytest.mark.parametrize("label,text", POSITIVE_KA_CASES, ids=[p[0] for p in POSITIVE_KA_CASES])
        def test_d05_lexicon_catches_planted_violation(label, text):
            r = check(text, locales=("ka",))
            assert not r.passed, f"{label}: expected violation, got {r}"
